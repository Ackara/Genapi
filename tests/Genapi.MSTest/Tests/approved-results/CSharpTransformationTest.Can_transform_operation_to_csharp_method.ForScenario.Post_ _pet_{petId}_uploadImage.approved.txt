﻿/// <summary>
/// uploads an image.
/// </summary>
// <param name="petId">ID of pet to update.</param>
public async Task<ApiResponse<ApiResponse>> uploadFileAsync(int petId)
{
	var request = new HttpRequestMessage(HttpMethod.Post, GetEndpoint($"/pet/{petId}/uploadImage"));
	HttpClient client = _factory.CreateClient();
	using (HttpResponseMessage response = await client.SendAsync(request))
	{
		switch ((int)response.StatusCode)
		{
			case 200:
				return new ApiResponse<ApiResponse>(JsonSerializer.Deserialize<ApiResponse>(await response.Content.ReadAsStringAsync(), _serializerOptions), (int)response.StatusCode);
		}
	}
}
